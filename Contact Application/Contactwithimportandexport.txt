    public static void ExportContact(Hashmap<String, Contact> ContactList) {
    	String Headder = "FullName DateOfBirth PrimaryNumber AdditionalNumber Group Notes SocialMedia";
    	File Contactfile = new File("C:\\Users\\deeve\\OneDrive\\Desktop\\File Handling\\ContactBook.csv");
		try {
			String additionalNumber = "",sociaMedia = "";
				FileWriter ContactBook = new FileWriter(Contactfile);
				ContactBook.append(Headder);
				for(Object contactObject : ContactList.getValue()) {
		    		Contact contact= (Contact) contactObject;
		    		if(contactObject != null) {
		    			ContactBook.append("\n");
		    			ContactBook.append(contact.firstName + contact.lastName + " ");
		    			ContactBook.append(contact.dateOfBirth + " ");
		    			ContactBook.append(contact.primaryMobileNo + " ");
		    			for(int index = 0;index < contact.AdditionalMobileNo.ArraySize();index++) {
		    	        	additionalNumber += contact.AdditionalMobileNo.get(index) ;
		    	        	if(index != contact.AdditionalMobileNo.ArraySize() - 1) {
		    	        		additionalNumber+=",";
		    	        	}
		    	        }
		    			ContactBook.append(additionalNumber + " ");
		    			ContactBook.append(contact.group + " ");
		    			ContactBook.append(contact.Notes + " ");
		    			for(Object key : contact.SocialNetworks.getKey()) {
		    				String keyString = (String) key;
		    				sociaMedia += keyString +":"+contact.SocialNetworks.get(keyString)+",";
		    			}
		    			ContactBook.append(sociaMedia);
		    		}
				}
				ContactBook.flush();
				ContactBook.close();
		   
		} catch (FileNotFoundException e) {
			e.printStackTrace();
		} catch (IOException e) {
			e.printStackTrace();
		}
	}

    	for(Object contactObject : ContactList.getValue()) {
